check prime number

code:

        const number = parseInt(prompt("Enter a positive number"));
        let isprime = true;

        if ( number === 1 ){
            alert("1 is neither prime nor composite number");
        }

        else if(number > 1)
        {
            for (let i = 2; i < number; i++){
                if(number % i == 0)
                {
                    isprime = false;
                    break;
                }
            }
            if(isprime)
            {
                alert(`${number} is a prime number`);
            }else {
                alert(`${number} is not a prime number`);
            }
        }
